// Get all resources.

// Get only level 1 resources with value = 2

  level: 1,
  value: 2
});
// Get only level 1 resources with value > 3.
var resources = Resources.get('resources', {
  level: {
    value: 1,
    //operator: '=',  // Default
  },
  value: {

    operator: '>',
  }
});
